function visao(tipo){ const painel = document.getElementById("painelVisao"); if(tipo==="profundidade"){ painel.innerHTML = "👁️ A percepção de profundidade permite estimar distâncias entre objetos."; } if(tipo==="movimento"){ painel.innerHTML = "🚗 O cérebro interpreta alterações de posição para perceber movimento."; } if(tipo==="velocidade"){ painel.innerHTML = "⚡ Quanto maior a variação da posição em menos tempo, maior a velocidade percebida."; } } function mover(personagem){ let elemento = document.getElementById(personagem); let pos = 20; let velocidade = 5; if(personagem==="onca"){ velocidade = 12; } if(personagem==="humano"){ velocidade = 8; } if(personagem==="cadeirante"){ velocidade = 6; } let anim = setInterval(()=>{ pos += velocidade; elemento.style.left = pos + "px"; if(pos > 800){ clearInterval(anim); } },30); const descricao = document.getElementById("descricaoMobilidade"); if(personagem==="cadeirante"){ descricao.innerHTML = "🧑‍🦽 Mobilidade assistida por cadeira de rodas."; } if(personagem==="onca"){ descricao.innerHTML = "🐆 A onça representa deslocamento rápido e agilidade."; } if(personagem==="humano"){ descricao.innerHTML = "🚶 Locomoção humana em velocidade moderada."; } }
